// ================================================================ // 社員スケジュール・日報管理 - Google Apps Script // このコードを全部コピーして Apps Script に貼り付けてください // ================================================================ const SHEET_REPORT = '日報'; const SHEET_SCHEDULE = 'スケジュール'; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const params = e.parameter; const action = params.action; let result; try { if (action === 'getReports') { result = getReports(params.date); } else if (action === 'saveReport') { result = saveReport(params); } else if (action === 'getSchedules') { result = getSchedules(params.date); } else if (action === 'saveSchedule') { result = saveSchedule(params); } else if (action === 'deleteSchedule') { result = deleteSchedule(params.id); } else { result = { status: 'error', message: 'Unknown action' }; } } catch (err) { result = { status: 'error', message: err.toString() }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function getSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); if (name === SHEET_REPORT) { sheet.appendRow(['ID','日付','氏名','作業内容','面接人数','工場見学人数','入社人数','明日の予定','達成度','課題','コメント','提出時間','元の作業内容','元の明日の予定','元の課題','元のコメント']); } else if (name === SHEET_SCHEDULE) { sheet.appendRow(['ID','日付','氏名','開始時間','終了時間','予定内容']); } } return sheet; } function getReports(date) { const sheet = getSheet(SHEET_REPORT); const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = data.slice(1).filter(r => r[1] === date && r[0] !== ''); const reports = rows.map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); return { status: 'ok', data: reports }; } function saveReport(p) { const sheet = getSheet(SHEET_REPORT); const data = sheet.getDataRange().getValues(); const now = new Date().toLocaleTimeString('ja-JP', {hour:'2-digit', minute:'2-digit'}); for (let i = 1; i < data.length; i++) { if (data[i][1] === p.date && data[i][2] === p.name) { sheet.getRange(i + 1, 1, 1, 16).setValues([[ data[i][0], p.date, p.name, p.work, +p.interview||0, +p.factory||0, +p.joined||0, p.next||'', +p.star||3, p.issue||'', p.comment||'', now, p.workOrig||'', p.nextOrig||'', p.issueOrig||'', p.commentOrig||'' ]]); return { status: 'ok', action: 'updated' }; } } const id = Date.now().toString(); sheet.appendRow([id, p.date, p.name, p.work, +p.interview||0, +p.factory||0, +p.joined||0, p.next||'', +p.star||3, p.issue||'', p.comment||'', now, p.workOrig||'', p.nextOrig||'', p.issueOrig||'', p.commentOrig||'']); return { status: 'ok', action: 'created' }; } function getSchedules(date) { const sheet = getSheet(SHEET_SCHEDULE); const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = data.slice(1).filter(r => r[1] === date && r[0] !== ''); const schedules = rows.map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); return { status: 'ok', data: schedules }; } function saveSchedule(p) { const sheet = getSheet(SHEET_SCHEDULE); const id = Date.now().toString(); sheet.appendRow([id, p.date, p.name, p.start, p.end, p.task]); return { status: 'ok', id: id }; } function deleteSchedule(id) { const sheet = getSheet(SHEET_SCHEDULE); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0].toString() === id.toString()) { sheet.deleteRow(i + 1); return { status: 'ok' }; } } return { status: 'error', message: 'Not found' }; }
